macro(add_plugin plugin)
  option(ENABLE_${plugin}_PLUGIN "Enable the ${plugin} plugin" ON)
  if(ENABLE_${plugin}_PLUGIN)
    add_subdirectory(${plugin})
  endif()
endmacro()

# ---------------------------------------------------------------------------- #

add_subdirectory(base)
